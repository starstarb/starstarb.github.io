完美数列
<!--more-->
```
#include<iostream>
#include<algorithm>
using namespace std;
const int MAXN=100010;
int a[MAXN];
int main()
{
	ios::sync_with_stdio(false);
	int p,n;
	cin>>n>>p;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	sort(a,a+n);
	int ans=1;
	for(int i=0;i<n;i++)
	{
		int j=upper_bound(a+i+1,a+n,(long long)a[i]*p)-a;
		ans=max(ans,j-i);
	}
	cout<<ans<<endl;
	return 0;
}
```
