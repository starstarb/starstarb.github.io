德才论
<!--more-->
```
#include <cstdio>
#include <algorithm>
using namespace std;
struct student{
    int id;
    int de,cai,sum;
    int rank;
}s[100010];
 
bool cmp(student a,student b){
    if(a.rank!=b.rank) return a.rank<b.rank;
    else if(a.sum!=b.sum) return a.sum>b.sum;
    else if(a.de!=b.de) return a.de>b.de;
    else return a.id<b.id;
}
int main(){
    int n,l,h,num;
    scanf("%d%d%d",&n,&l,&h);//考生总数 最低分  优先录取线 
    num=n;//及格人数 
    for(int i=0;i<n;i++){
        scanf("%d%d%d",&s[i].id,&s[i].de,&s[i].cai);
        s[i].sum=s[i].de+s[i].cai;
        if(s[i].de<l ||s[i].cai<l){//不及格设为第五类 
            num--;//及格人数减一 
            s[i].rank=5;
        }
        else{
            if(s[i].de>=h && s[i].cai>=h) s[i].rank=1;
            else if(s[i].de>=h && s[i].cai<h) s[i].rank=2;
            else if(s[i].de>=s[i].cai) s[i].rank=3;
            else s[i].rank=4;
        }
    }
    sort(s,s+n,cmp);
    printf("%d\n",num);
    for(int i=0;i<num;i++)
        printf("%d %d %d\n",s[i].id,s[i].de,s[i].cai);
 
    return 0;
}
```
