完美数列
<!--more-->
```
#include<iostream>
#include<algorithm>
using namespace std;
const int MAXN=100010;
int main()
{
	ios::sync_with_stdio(false);
    int n,p,a[MAXN];
    cin>>n>>p;
    for(int i=0;i<n;i++){
    	cin>>a[i];
	}
	sort(a,a+n);//从小到大排序 
	int i=0,j=0,flag=1;//双指针   
	while(i<n&&j<n){
		while(j<n&&a[j]<=(long long)a[i]*p){
			flag=max(flag,j-i+1);
			j++;//j右移 
		}
		i++;//保证i和j的距离最大 
	}
	cout<<flag<<endl;
	return 0;
}
```
