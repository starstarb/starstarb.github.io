完美数列
<!--more-->
```
#include<iostream>
#include<algorithm>
using namespace std;
const int MAXN=100010;
int a[MAXN],n;
int binarySearch(int i,long long x){
	if(a[n-1]<=x) return n;
	int l=i+1,r=n-1,mid;
	while(l<r)
	{
		mid=(l+r)/2;
		if(a[mid]<=x) l=mid+1;
		else r=mid;
	}
	return l;
}
int main()
{
	ios::sync_with_stdio(false);
	int p;
	cin>>n>>p;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	sort(a,a+n);
	int ans=1;
	for(int i=0;i<n;i++)
	{
		int j=binarySearch(i,a[i]*p);//在a[i+1]~a[n-1]中查找第一个超过a[i]*p的数，返回位置给j 
		ans=max(ans,j-i);
	}
	cout<<ans<<endl;
	return 0;
}
```
